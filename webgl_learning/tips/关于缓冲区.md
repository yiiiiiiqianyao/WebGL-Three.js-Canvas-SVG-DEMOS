以下是关于缓冲的使用和简单说明

```javascript
let buffer = gl.createBuffer() 
// 创建缓冲区对象

gl.bindBuffer(gl.ARRAY_BUFFER, buffer) 
// 将缓冲区对象绑定到目标 gl.ARRAY_BUFFER 是一个 webgl 全局可索引的唯一对象

const vertices = new Float32Array([
    ...
])
gl.bufferData(gl.ARRAY_BUFFER, vertices, gl.STATIC_DRAW) 
// 向当前绑定在 gl.ARRAY_BUFFER 索引对象的缓冲区中写入数据 vertices、gl.STATIC_DRAW 告诉 webgl 将如何使用缓冲区中的数据

let attr = gl.getAttribLocation(program, attrName);
// 获取变量地址

gl.vertexAttribPointer(attr, count, gl.FLOAT, false, 0, 0);
// 将当前绑定在 gl.ARRAY_BUFFER 中的数据按照一定的步进分配给变量 attr

gl.enableVertexAttribArray(attr);
// 开启变量 attr
```
